const SHEET_NAME = 'Technicians Toolkits'; const PARTS_LIST = [ ['White Solenoid Valve', 1], ['Black Solenoid Valve', 2], ['Solenoid Repair Kit', 4], ['Actuator', 2], ['Gas Gauge — Male', 1], ['Gas Gauge — Female', 1], ['Secondary Gas Gauge', 1], ['3/16" Pipe (metres)', 2], ['5/16" Pipe (metres)', 1], ['3/8" Pipe (metres)', 2], ['1/2" Pipe (metres)', 0], ['15mm Pipe (metres)', 0], ['Lagging (metres)', 1], ['S-Type Coupler', 1], ['A-Type Coupler', 1], ['G-Type Coupler', 1], ['D-Type Coupler', 1], ['U-Type Coupler', 1], ['Non-Return Valve', 0], ['Equal Straight — 3/16"', 3], ['Equal Straight — 5/16"', 3], ['Equal Straight — 3/8"', 3], ['Equal Straight — 1/2"', 3], ['Elbow — 3/8"', 3], ['Elbow — 15mm', 3], ['Elbow — 18mm', 3], ['Reducer — 5/16" to 3/16"', 6], ['Reducer — 3/8" to 5/16"', 3], ['Y Piece', 4], ['Brass Coupler Fitting (Grey)', 2], ['Brass Coupler Fitting (White)', 2], ['Blanking Cap', 2], ['Tap', 2], ['Nozzle (Standard)', 4], ['Nozzle (Guinness)', 4], ]; function getOrMakeSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var tab = ss.getSheetByName(SHEET_NAME); if (!tab) { tab = ss.insertSheet(SHEET_NAME); tab.getRange(1, 1).setValue('Part').setFontWeight('bold').setBackground('#f3f3f3'); tab.getRange(1, 2).setValue('Recommended').setFontWeight('bold').setBackground('#f3f3f3'); tab.setColumnWidth(1, 220); tab.setColumnWidth(2, 120); tab.setFrozenRows(1); tab.setFrozenColumns(2); // Write all parts in one batch call var partData = PARTS_LIST.map(function(p) { return [p[0], p[1]]; }); tab.getRange(2, 1, partData.length, 2).setValues(partData); tab.getRange(2, 2, partData.length, 1).setBackground('#f3f3f3'); } return tab; } function addTechnicianColumn(tab, tech) { var newCol = tab.getLastColumn() + 1; tab.setColumnWidth(newCol, 120); tab.getRange(1, newCol).setValue(tech).setFontWeight('bold') .setBackground('#E91E8C').setFontColor('#ffffff'); var numRows = PARTS_LIST.length; // Write all zeros in one call var zeros = PARTS_LIST.map(function() { return [0]; }); tab.getRange(2, newCol, numRows, 1).setValues(zeros); // Read recommended quantities in one call then colour all at once var recs = tab.getRange(2, 2, numRows, 1).getValues(); var backgrounds = recs.map(function(r) { return [applyColourValue(0, Number(r[0]) || 0)]; }); tab.getRange(2, newCol, numRows, 1).setBackgrounds(backgrounds); return newCol; } function doGet(e) { var action = (e.parameter && e.parameter.action) || ''; var tab = getOrMakeSheet(); if (action === 'getTechnicians') { var lastCol = tab.getLastColumn(); if (lastCol < 3) return json({technicians: []}); // Read all header row in one call var headers = tab.getRange(1, 3, 1, lastCol - 2).getValues()[0]; var techs = headers.filter(function(n) { return String(n).trim(); }) .map(function(n) { return String(n).trim(); }); return json({technicians: techs}); } if (action === 'getToolkit') { var tech = String(e.parameter.technician || '').trim(); var col = findTechCol(tab, tech); if (col === -1) col = addTechnicianColumn(tab, tech); var numRows = PARTS_LIST.length; // Read parts, recommended AND tech quantities all in one batch var allData = tab.getRange(2, 1, numRows, col).getValues(); var parts = {}, std = {}; for (var r = 0; r < allData.length; r++) { var part = String(allData[r][0]).trim(); if (!part) continue; std[part] = Number(allData[r][1]) || 0; parts[part] = Number(allData[r][col - 1]) || 0; } return json({parts: parts, std: std}); } return ContentService.createTextOutput('OK'); } function doPost(e) { try { var data = JSON.parse(e.postData.contents); var tab = getOrMakeSheet(); if (data.action === 'registerTechnician') { var tech = String(data.technician || '').trim(); if (!tech) return ContentService.createTextOutput('No name'); if (findTechCol(tab, tech) === -1) addTechnicianColumn(tab, tech); return ContentService.createTextOutput('OK'); } if (data.action === 'updateQuantities') { var tech = String(data.technician || '').trim(); var col = findTechCol(tab, tech); if (col === -1) col = addTechnicianColumn(tab, tech); var quantities = data.quantities || {}; var numRows = PARTS_LIST.length; // Read all data in one call var allData = tab.getRange(2, 1, numRows, col).getValues(); var newVals = [], newBgs = []; for (var r = 0; r < allData.length; r++) { var part = String(allData[r][0]).trim(); var rec = Number(allData[r][1]) || 0; var qty = quantities[part] !== undefined ? Math.max(0, Number(quantities[part])) : Number(allData[r][col-1]) || 0; newVals.push([qty]); newBgs.push([applyColourValue(qty, rec)]); } tab.getRange(2, col, numRows, 1).setValues(newVals); tab.getRange(2, col, numRows, 1).setBackgrounds(newBgs); return ContentService.createTextOutput('OK'); } if (data.action === 'deductParts') { var tech = String(data.technician || '').trim(); var col = findTechCol(tab, tech); if (col === -1) return ContentService.createTextOutput('Not found'); var parts = data.parts || {}; var numRows = PARTS_LIST.length; var allData = tab.getRange(2, 1, numRows, col).getValues(); var newVals = [], newBgs = []; for (var r = 0; r < allData.length; r++) { var part = String(allData[r][0]).trim(); var rec = Number(allData[r][1]) || 0; var current = Number(allData[r][col-1]) || 0; var qty = parts[part] ? Math.max(0, current - Number(parts[part])) : current; newVals.push([qty]); newBgs.push([applyColourValue(qty, rec)]); } tab.getRange(2, col, numRows, 1).setValues(newVals); tab.getRange(2, col, numRows, 1).setBackgrounds(newBgs); return ContentService.createTextOutput('OK'); } } catch(err) { return ContentService.createTextOutput('Error: ' + err.message); } return ContentService.createTextOutput('OK'); } function onEdit(e) { var sheet = e.range.getSheet(); if (sheet.getName() !== SHEET_NAME) return; var row = e.range.getRow(); var col = e.range.getColumn(); if (col < 3 || row < 2) return; var qty = Number(e.range.getValue()) || 0; var rec = Number(sheet.getRange(row, 2).getValue()) || 0; e.range.setBackground(applyColourValue(qty, rec)); } function findTechCol(tab, name) { var lastCol = tab.getLastColumn(); if (lastCol < 3) return -1; var headers = tab.getRange(1, 3, 1, lastCol - 2).getValues()[0]; for (var i = 0; i < headers.length; i++) { if (String(headers[i]).trim().toLowerCase() === name.trim().toLowerCase()) return i + 3; } return -1; } // Returns colour string instead of setting it — allows batch setBackgrounds function applyColourValue(qty, rec) { if (qty === 0) return '#f4cccc'; if (rec > 0 && qty < rec) return '#fce5cd'; return '#d9ead3'; } function applyColour(cell, qty, rec) { cell.setBackground(applyColourValue(qty, rec)); } function json(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
